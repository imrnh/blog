<script>
  fetch('research_projects.txt')
    .then(response => response.text())
    .then(text => {
        const items = parseText(text);
        displayCards(items);
    })
  .catch(error => console.error('Error:', error));

  function parseText(text) {
      const items = [];
      let currentItem = {};

      text.split('\n').forEach(line => {
          if (line.trim() === '') {
              if (Object.keys(currentItem).length > 0) {
                  items.push(currentItem);
                  currentItem = {};
              }
          } else {
              const [key, value] = line.split(': ');
              if (key && value) {
                  currentItem[key.trim()] = value.trim();
              }
          }
      });

      if (Object.keys(currentItem).length > 0) {
          items.push(currentItem);
      }

      return items;
  }

  function displayCards(items) {
      const container = document.getElementById('cardContainer');
      container.innerHTML = '';

      items.forEach(item => {
          const card = document.createElement('div');
          card.className = 'card';
          card.innerHTML = `
              <h3>${item.Name}</h3>
              <p>${item.Description}</p>
          `;
          container.appendChild(card);
      });
  }

  // Initialize with sample data
  const sampleData = `Name: Hopfield Network
    Description: This blog is about human memory.

    Name: Graph Neural Network
    Description: Used to find connection between 2 node.`;

  const items = parseText(sampleData);
  displayCards(items);
</script>
